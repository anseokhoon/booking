/**************************************************************** * KB금융 스타웰스매니저 상담예약 — Apps Script (A안 / primary 전용) * * 운영 원칙 * - 이 Apps Script는 반드시 캘린더 주인 계정으로 로그인해 배포해야 합니다. * (예: seokhoonbaro@gmail.com) * - 서비스 계정 / OAuth 클라이언트 / API 키 사용하지 않습니다. * - 배포 계정의 primary 캘린더만 읽고 씁니다. * * 배포 순서 * 1) seokhoonbaro@gmail.com 으로 로그인 * 2) https://script.google.com 접속 후 프로젝트 열기 * 3) Code.gs 전체 교체 후 저장 * 4) 배포 → 새 배포 → 웹 앱 * - 실행 사용자: 나 (배포 계정) * - 액세스 권한: 모든 사용자 * 5) 권한 동의 시 반드시 캘린더 주인 계정으로 승인 * 6) 발급된 .../exec URL 을 index.html 의 CONFIG.WEB_APP_URL 에 반영 * * 진단 URL * - .../exec?action=ping * - .../exec?action=health * - .../exec?action=list&from=2026-05-13 ****************************************************************/ const CONFIG = { // A안: 항상 primary 고정 CALENDAR_MODE: "primary", // 캘린더 주인 계정 (반드시 이 계정으로 Apps Script 로그인 & 배포) EXPECTED_OWNER_EMAIL: "[email protected]", // 매니저 정보 MANAGER_NAME: "안석훈", MANAGER_EMAIL: "[email protected]", MANAGER_PHONE: "010-2213-3907", // 운영 정책 OPEN_HOUR: 9, CLOSE_HOUR: 20, DURATION_HOUR: 1, BUFFER_HOUR: 1, WORK_DAYS: [1, 2, 3, 4, 5, 6], // 월~토 (0=일) TIMEZONE: "Asia/Seoul", // 예약 가능 기간 ADVANCE_DAYS: 1, MAX_AHEAD_DAYS: 60 }; /**************************************************************** * GET ****************************************************************/ function doGet(e) { try { const action = (e && e.parameter && e.parameter.action) || "list"; if (action === "ping") { return jsonOut_({ success: true, message: "pong", deployed_account_hint: Session.getActiveUser().getEmail() || null, timestamp: nowStr_() }); } if (action === "health") { return jsonOut_(getHealth_()); } if (action === "list") { const from = (e.parameter.from) || todayStr_(); const slots = listBlockedSlots_(from); return jsonOut_({ success: true, slots: slots }); } return jsonOut_({ success: false, error: "Unknown action" }); } catch (err) { return jsonOut_({ success: false, error: String(err && err.message || err) }); } } /**************************************************************** * POST — 예약 등록 ****************************************************************/ function doPost(e) { try { let action, payload; if (e.parameter && e.parameter.action) { action = e.parameter.action; payload = e.parameter.payload ? JSON.parse(e.parameter.payload) : {}; } else if (e.postData && e.postData.contents) { const obj = JSON.parse(e.postData.contents); action = obj.action; payload = obj.payload || obj; } else { throw new Error("요청 데이터를 읽을 수 없습니다"); } if (action === "book") return jsonOut_(handleBooking_(payload)); return jsonOut_({ success: false, error: "Unknown action" }); } catch (err) { return jsonOut_({ success: false, error: String(err && err.message || err) }); } } /**************************************************************** * 헬스 체크 — A안 핵심 ****************************************************************/ function getHealth_() { try { const cal = getPrimaryCalendar_(); const ownerEmail = cal.getId(); // primary는 보통 계정 이메일 const now = new Date(); const upcoming = cal.getEvents(now, new Date(now.getTime() + 24 * 3600 * 1000)).length; const matched = ownerEmail && ownerEmail.toLowerCase() === CONFIG.EXPECTED_OWNER_EMAIL.toLowerCase(); return { success: true, calendar_mode: "primary", calendar_id: ownerEmail, calendar_name: cal.getName(), expected_owner: CONFIG.EXPECTED_OWNER_EMAIL, owner_match: matched, upcoming_events_24h: upcoming, timezone: CONFIG.TIMEZONE, timestamp: nowStr_(), hint: matched ? "정상: 배포 계정 == 캘린더 주인 계정" : "경고: 현재 Apps Script 로그인 계정이 캘린더 주인 계정과 다릅니다. 캘린더 주인 계정으로 재배포 필요." }; } catch (err) { return { success: false, calendar_mode: "primary", error: String(err && err.message || err), hint: "캘린더 주인 계정(seokhoonbaro@gmail.com)으로 Apps Script 로그인 후 새로 배포해주세요.", timestamp: nowStr_() }; } } /**************************************************************** * 예약 처리 ****************************************************************/ function handleBooking_(p) { if (!p || !p.name || !p.phone || !p.topic || !p.date || !p.time) { return { success: false, error: "필수 항목이 누락되었습니다" }; } if (!/^010-\d{3,4}-\d{4}$/.test(p.phone)) { return { success: false, error: "연락처 형식이 올바르지 않습니다" }; } if (!/^\d{4}-\d{2}-\d{2}$/.test(p.date) || !/^\d{2}:\d{2}$/.test(p.time)) { return { success: false, error: "날짜/시간 형식이 올바르지 않습니다" }; } const [hh] = p.time.split(":").map(Number); if (getCandidateHours_().indexOf(hh) < 0) { return { success: false, error: "선택할 수 없는 시간입니다" }; } const startDate = parseLocal_(p.date, p.time); if (CONFIG.WORK_DAYS.indexOf(startDate.getDay()) < 0) { return { success: false, error: "운영하지 않는 요일입니다" }; } if (startDate.getTime() < Date.now() - 60 * 1000) { return { success: false, error: "이미 지난 시각은 예약할 수 없습니다" }; } const endDate = new Date(startDate.getTime() + CONFIG.DURATION_HOUR * 3600 * 1000); const cal = getPrimaryCalendar_(); const protectedStart = new Date(startDate.getTime() - CONFIG.BUFFER_HOUR * 3600 * 1000); const protectedEnd = new Date(endDate.getTime() + CONFIG.BUFFER_HOUR * 3600 * 1000); const conflicts = cal.getEvents(protectedStart, protectedEnd); if (conflicts.length > 0) { return { success: false, error: "기존 일정 또는 예약과 겹칩니다. 다른 시간을 선택해주세요." }; } const title = "[KB상담] " + p.name + " · " + p.topic; const description = "▣ KB금융 스타웰스매니저 상담 예약\n" + "──────────────────────────\n" + "▪ 고객명: " + p.name + "\n" + "▪ 연락처: " + p.phone + "\n" + "▪ 이메일: " + (p.email || "(미입력)") + "\n" + "▪ 상담분야: " + p.topic + "\n" + "▪ 요청메시지: " + (p.memo || "(없음)") + "\n" + "──────────────────────────\n" + "▪ 담당: " + CONFIG.MANAGER_NAME + " (" + CONFIG.MANAGER_PHONE + ")\n" + "▪ 접수일시: " + nowStr_(); const event = cal.createEvent(title, startDate, endDate, { description: description, location: "고객 내방 또는 고객 계시는 곳 방문 안내" }); event.setTag("kb-booking", "1"); event.setTag("customer", p.name); event.setTag("phone", p.phone); try { MailApp.sendEmail( CONFIG.MANAGER_EMAIL, "[KB상담예약] " + p.name + " · " + p.date + " " + p.time, "새로운 상담 예약이 접수되었습니다.\n\n" + "고객명: " + p.name + "\n" + "연락처: " + p.phone + "\n" + "이메일: " + (p.email || "-") + "\n" + "분야: " + p.topic + "\n" + "일시: " + p.date + " " + p.time + " ~ " + pad_(hh + 1) + ":00\n" + "메시지: " + (p.memo || "-") + "\n\n" + "캘린더에 자동 등록되었습니다." ); } catch (mailErr) { Logger.log("매니저 메일 실패: " + mailErr); } if (p.email && /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(p.email)) { try { MailApp.sendEmail( p.email, "[KB금융] 상담 예약이 접수되었습니다 - " + CONFIG.MANAGER_NAME + " 매니저", p.name + "님, 안녕하세요.\n\n" + "KB금융 스타웰스매니저 " + CONFIG.MANAGER_NAME + "입니다.\n" + "아래와 같이 상담 예약이 접수되었습니다.\n\n" + "▪ 일시: " + p.date + " " + p.time + " ~ " + pad_(hh + 1) + ":00\n" + "▪ 분야: " + p.topic + "\n" + "▪ 장소: 고객 내방 또는 고객 계시는 곳 방문 안내\n\n" + "확정 안내는 별도로 연락드리겠습니다.\n" + "문의: " + CONFIG.MANAGER_PHONE + "\n\n감사합니다." ); } catch (mailErr) { Logger.log("고객 메일 실패: " + mailErr); } } return { success: true, eventId: event.getId(), date: p.date, time: p.time }; } /**************************************************************** * 차단된 슬롯 목록 — 기존 일정 + 예약 + 버퍼 반영 ****************************************************************/ function listBlockedSlots_(fromDateStr) { const cal = getPrimaryCalendar_(); const from = parseLocal_(fromDateStr, "00:00"); from.setHours(0, 0, 0, 0); const to = new Date(from.getTime() + 65 * 24 * 3600 * 1000); const events = cal.getEvents(from, to); const result = {}; for (let day = new Date(from.getTime()); day < to; day.setDate(day.getDate() + 1)) { if (CONFIG.WORK_DAYS.indexOf(day.getDay()) < 0) continue; const dateKey = Utilities.formatDate(day, CONFIG.TIMEZONE, "yyyy-MM-dd"); const blocked = []; getCandidateHours_().forEach(function (hour) { const slotStart = new Date(day.getFullYear(), day.getMonth(), day.getDate(), hour, 0, 0, 0); const slotEnd = new Date(slotStart.getTime() + CONFIG.DURATION_HOUR * 3600 * 1000); const protectedStart = new Date(slotStart.getTime() - CONFIG.BUFFER_HOUR * 3600 * 1000); const protectedEnd = new Date(slotEnd.getTime() + CONFIG.BUFFER_HOUR * 3600 * 1000); const hasConflict = events.some(function (ev) { return overlaps_(ev.getStartTime(), ev.getEndTime(), protectedStart, protectedEnd); }); if (hasConflict) blocked.push(pad_(hour) + ":00"); }); if (blocked.length > 0) result[dateKey] = blocked; } return result; } /**************************************************************** * 유틸 ****************************************************************/ function getPrimaryCalendar_() { const cal = CalendarApp.getDefaultCalendar(); if (!cal) { throw new Error("primary 캘린더를 찾을 수 없습니다. 캘린더 주인 계정으로 Apps Script 로그인 후 재배포해주세요."); } return cal; } function getCandidateHours_() { const hours = []; for (let h = CONFIG.OPEN_HOUR; h + CONFIG.DURATION_HOUR <= CONFIG.CLOSE_HOUR; h += CONFIG.DURATION_HOUR) { hours.push(h); } return hours; } function overlaps_(aStart, aEnd, bStart, bEnd) { return aStart.getTime() < bEnd.getTime() && aEnd.getTime() > bStart.getTime(); } function parseLocal_(dateStr, timeStr) { const [y, m, d] = dateStr.split("-").map(Number); const [h, mi] = timeStr.split(":").map(Number); return new Date(y, m - 1, d, h, mi, 0, 0); } function jsonOut_(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function pad_(n) { return (n < 10 ? "0" : "") + n; } function todayStr_() { return Utilities.formatDate(new Date(), CONFIG.TIMEZONE, "yyyy-MM-dd"); } function nowStr_() { return Utilities.formatDate(new Date(), CONFIG.TIMEZONE, "yyyy-MM-dd HH:mm:ss"); }
